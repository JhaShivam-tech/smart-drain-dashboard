Firebase Realtime Database Integration

Add the following code to connect live sensor data from your IoT devices:

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
import { getDatabase, ref, onValue } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-database.js";

const firebaseConfig = {
    apiKey: "AIzaSy...YOUR_KEY",
    databaseURL: "https://smartdrain-mumbai-default-rtdb.firebaseio.com"
};

const app = initializeApp(firebaseConfig);
const db = getDatabase(app);

// Listen to drains node in realtime
const drainsRef = ref(db, "drains");
onValue(drainsRef, (snapshot) => {
    const data = snapshot.val();
    console.log("🔥 Realtime drain data received:", data);
    // updateAllDrains(data);
});
